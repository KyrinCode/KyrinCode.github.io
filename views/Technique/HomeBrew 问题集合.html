      Kyrin 2020-03-09 mac brew

      Mac 的包管理利器，记录一些使用中常用的功能。

      # Homebrew 使用

      # 查看版本
brew -v
# 下载软件
brew install <software>
# 需要图形安装的软件安装
brew cask install <software>
# 卸载软件
brew uninstall <software>
brew cask uninstall <software>
# 软件搜索（
brew search <software>
brew search /^vi/ # 支持*正则表达式*：（搜索以vi开头的软件）
# 列出已安装的包
brew list
brew list --cask
# 列出过时的包
brew outdated
# 查看软件相关信息
brew info <software>
# 删除下载的包
brew cleanup
# 更新 Homebrew
brew update

# Homebrew 扩展仓库
# 列出已有仓库
brew tap
# 添加仓库
brew tap <repository>
# 删除仓库
brew untap <repository>

      # HomeBrew 换源

      mac 的 包管理软件 HomeBrew 用官方源总是很慢，换国内源解决问题

      # 切换到 Homebrew 目录

      cd "$(brew --repo)"
      1

      # 替换brew.git

      git remote -v #查看远程仓库，默认的使用的github
git remote rm origin #删除远程

# 阿里源
git remote add origin https://mirrors.aliyun.com/homebrew/brew.git #添加
git remote set-url origin https://mirrors.aliyun.com/homebrew/brew.git #切换
# 科大源
git remote add origin https://mirrors.ustc.edu.cn/brew.git
git remote set-url origin https://mirrors.ustc.edu.cn/brew.git
# 清华源
git remote add origin https://mirrors.tuna.tsinghua.edu.cn/git/homebrew/brew.git
git remote set-url origin https://mirrors.tuna.tsinghua.edu.cn/git/homebrew/brew.git

      # 切换回官方源

      # 重置brew.git
cd "$(brew --repo)"
git remote set-url origin https://github.com/Homebrew/brew.git

# 重置homebrew-core.git
cd "$(brew --repo)/Library/Taps/homebrew/homebrew-core"
git remote set-url origin https://github.com/Homebrew/homebrew-core.git

# 重置homebrew-cask.git
cd "$(brew --repo)"/Library/Taps/homebrew/homebrew-cask
git remote set-url origin https://github.com/caskroom/homebrew-cask.git

# 重置homebrew-bottles
vi ~/.bash_profile
# 删除HOMEBREW_BOTTLE_DOMAIN这一行配置
source ~/.bash_profile
